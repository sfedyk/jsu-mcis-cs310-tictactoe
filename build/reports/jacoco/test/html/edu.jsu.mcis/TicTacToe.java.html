	public boolean getPlayer(int gameCount){
		if(gameCount % 2 == 0)
			player1 = true;
		else
			player1 = false;
		return player1;
	}

	public void actionPerformed(ActionEvent event){
		TicTacToe game = new TicTacToe();
		String [] moves = new String[9];
		
		if (event.getSource() == button1){
			if(game.getPlayer(count) == true){
				button1.setText("x");
				moves[0] = "X";
				count ++;
			}
			else
			{
				button1.setText("0");
				moves[0] = "O";
				count ++;
			}
				
		}

		if (event.getSource() == button2){
			if(game.getPlayer(count) == true){
				button2.setText("x");
				moves[1] = "X";
				count ++;
			}
			else
			{
				button2.setText("0");
				moves[1] = "O";
				count ++;
			}
		
		}
		if (event.getSource() == button3){
			if(game.getPlayer(count) == true){
				button3.setText("x");
				moves[2] = "X";
				count ++;
			}
			else
			{
				button3.setText("0");
				moves[2] = "O";
				count ++;
			}
			
		}
		if (event.getSource() == button4){
			if(game.getPlayer(count) == true){
				button4.setText("x");
				moves[3] = "X";
				count ++;
			}
			else
			{
				button4.setText("0");
				moves[3] = "O";
				count ++;
			}
		
		}
		if (event.getSource() == button5){
			if(game.getPlayer(count) == true){
				button5.setText("x");
				moves[4] = "X";
				count ++;
			}
			else
			{
				button5.setText("0");
				moves[4] = "O";
				count ++;
			}
		
		}
		if (event.getSource() == button6){
			if(game.getPlayer(count) == true){
				button6.setText("x");
				moves[5] = "X";
				count ++;
			}
			else
			{
				button6.setText("0");
				moves[5] = "O";
				count ++;
			}
						
		}
		if (event.getSource() == button7){
			if(game.getPlayer(count) == true){
				button7.setText("x");
				moves[6] = "X";
				count ++;
			}
			else
			{
				button7.setText("0");
				moves[6] = "O";
				count ++;
			}
						
		}
		if (event.getSource() == button8){
			if(game.getPlayer(count) == true){
				button8.setText("x");
				moves[7] = "X";
				count ++;
			}
			else
			{
				button8.setText("0");
				moves[7] = "O";
				count ++;
			}
						
		}
		if (event.getSource() == button9){
			if(game.getPlayer(count) == true){
				button9.setText("x");
				moves[8] = "X";
				count ++;
			}
			else
			{
				button9.setText("0");
				moves[8] = "O";
				count ++;
			}
						
		}		
		
		
		if(game.checkWinner(moves) == "X"){
				winner.setText("X WINS");
			}
		else if(game.checkWinner(moves) == "O"){
				winner.setText("O WINS");
			}
		else
			winner.setText("Tie");
		
	}
